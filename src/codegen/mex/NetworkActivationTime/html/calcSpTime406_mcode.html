1function spTime = calcSpTime(i,nodes,x)
 2    spTime = 0.0; nAct = 0; spTimePrev = 0.0;
 3    for j = 1:nodes(i).par.n
 4        spTime = spTime + double(nodes(nodes(i).neigh(j)).x == x) / nodes(i).w(j); % summing up the inverse weights (sp times) of active neighbors to reach i
 5        spTimePrev = spTimePrev + double(nodes(nodes(i).neigh(j)).x==x) * nodes(nodes(i).neigh(j)).par.spTime; % summing up the sp time to reach each of the active neighbors
 6        nAct = nAct + double(nodes(nodes(i).neigh(j)).x==x); % x is either 0 or 1, so nAct is the total amount of active neighbors of i
 7    end
 8    if (nAct == 0)
 9        nAct = 1;
10    end
11    spTime = (spTime + spTimePrev) / nAct;
12end
