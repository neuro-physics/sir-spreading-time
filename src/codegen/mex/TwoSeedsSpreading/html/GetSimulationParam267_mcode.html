1function p = GetSimulationParam(A, nSeeds, isCompetitive, seedType, seedInd, nSim)
 2% A -> adj matrix
 3% nSeeds -> amount of initial seeds for activity (must be equal to the GetNodes nSeeds
 4% isCompetitive -> determines whether the seeds are competitive (diff states) or cooperative (eq states)
 5% seedType -> 'random': randomly select seeds
 6%             'fixed': select fixed seedInd for every trial + a quantity of (nSeeds-numel(seedInd)) seeds in 1:size(A,1);
 7%             'mostconn': selects the nSeeds most connected nodes as seeds
 8% seedInd -> double(1:nSim,1:nSeeds) index of 'fixed' seeds
 9%            if there is less rows than nSim, the remaining rows are a periodic repetition of seedInd(:,:)
10%            if there is less columns than nSeeds, the remaining cols are going to be filled up with random seeds
11% nSim -> number seed trials
12%
13% return:
14% p struct with fields:
15% .N -> network total number of elements
16% .nSeeds -> same as input
17% .isCompetitive -> same as input
18% .seeds -> double(1:nSim,1:nSeeds); each line is a seed trial
19% .seedTrial -> a counter for sim functions to determine which seed trial (out of .seeds) to take
20% .nSim -> number of seed trials
21% .seedType -> same as input
22% .x0 -> double(1,1:nSeeds); initial condition for each of the seed nodes
23%
24    if nargin == 0
25        p.N = 0;
26        p.nSeeds = 0;
27        p.isCompetitive = false;
28        p.seeds = 0;
29        p.seedTrial = 0;
30        p.nSim = 0;
31        p.seedType = '';
32        p.x0 = 0;
33        return;
34    end
35    N = size(A,1);
36    p.N = N;
37    p.nSeeds = nSeeds;
38    p.isCompetitive = isCompetitive;
39    p.seeds = zeros(nSim, nSeeds);
40    p.nSim = nSim;
41    p.seedType = seedType;
42    if strcmpi(seedType, 'random')
43        % generate random permutations...
44        % need to take care with the total amount of unique combinations of N, take nSeeds;
45        % s = N!/( (nSeeds)! * (N-nSeeds)!) is total number of unique combinations
46        % if s < nSim, then we can only take s maximum seed combinations, instead of nSim
47        %    otherwise we will sample with replacement
48        for i = 1:nSim
49            p.seeds(i,:) = randperm(N, nSeeds);
50        end
51    elseif strcmpi(seedType, 'fixed')
52        p.seeds = fillUpMatrix(seedInd,nSim,nSeeds,N);
53    elseif strcmpi(seedType, 'mostconn')
54        s = sort(sum(A~=0), 'descend');
55        s = s(1:nSeeds); % gets nSeeds most connected elements for every simulation
56        p.seeds = repmat(s(:)', nSim, 1);
57    else
58        error('GetSimulationParam:seedType', 'Unrecognized seed type');
59    end
60    p.seedTrial = 1;
61    if isCompetitive
62        p.x0 = 1:nSeeds;
63    else
64        p.x0 = ones(1,nSeeds);
65    end
66end
