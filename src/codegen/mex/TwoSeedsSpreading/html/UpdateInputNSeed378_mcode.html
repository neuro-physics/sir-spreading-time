1function [Isyn,kMax] = UpdateInputNSeed(i, nodes)
 2    %Isyn = zeros(1,nodes(i).par.nInp);
 3    IsynM = 0.0;
 4    kMax = 1;
 5    IsynAnt = -Inf;
 6    Isyn = IsynM;
 7    for k = 1:nodes(i).par.nInp
 8        % summing up contribution of color k
 9        IsynM = 0.0;
10        for j = 1:nodes(i).par.n
11            IsynM = IsynM + nodes(i).w(j) .* (nodes(nodes(i).neigh(j)).x == k);
12        end
13        
14        % selecting color of largest contribution
15        if IsynM > IsynAnt
16            kMax = k;
17            Isyn = IsynM;
18        elseif IsynM == IsynAnt
19            if rand > 0.5
20                kMax = k;
21                Isyn = IsynM;
22            end
23        end
24        IsynAnt = IsynM;
25    end
26end
