1function [T,V,S] = SpreadingTime(A, Theta, nSim, tTotal)
 2    nodes = GetNodes(A, Theta, 0, 1, false);
 3    N = numel(nodes);
 4    T = zeros(N,N);
 5    V = zeros(N,N);
 6    S = zeros(N,N);
 7    for i = 1:N
 8        for j = 1:N
 9            [T(i,j), V(i,j), S(i,j)] = SpreadNodeToNodeNTimes(i, j, nodes, nSim, tTotal);
10        end
11    end
12end

14function [T,V,S] = SpreadNodeToNodeNTimes(i, j, nodes, nSim, tTotal)
15    T = 0.0;
16    S = 0.0;
17    V = 0.0;
18    nn = 0;
19    for k = 1:nSim
20        [TT,SS] = SpreadNodeToNode(i,j,nodes,tTotal);
21        if ~isnan(TT)
22            T = T + TT;
23            V = V + TT .* TT;
24            S = S + SS;
25            nn = nn + 1;
26        end
27    end
28    if nn == 0
29        % all TT were NaN
30        T = NaN;
31        V = NaN;
32        S = NaN;
33    else
34        T = T ./ nn;
35        V = V ./ nn - T .* T;
36        S = S ./ nn;
37    end
38    if V < 0.0
39        V = 0.0;
40    end
41end
