1function T = findLinearTransformation(u,v)
 2% finds a matrix T (3x3) such that T * u(:,i) = v(:,i)
 3% u(:,i) and v(:,i) are 3d col vectors; the amount of col vectors in u and v must match
 4%
 5% explanation of algorithm in:
 6% http://stackoverflow.com/questions/26635266/matlab-calculate-3d-similarity-transformation-fitgeotrans-for-3d
 7    if any(size(u) ~= size(v))
 8        error('findLinearTransform:u','u and v must be the same shape and size 3d vectors');
 9    end
10    [n,nP] = size(u);
11    if nP > n
12        I = eye(nP);
13        u = [u;I((n+1):nP,1:nP)];
14        v = [v;I((n+1):nP,1:nP)];
15        [n,nP] = size(u);
16    end
17    if (nP*n*n*n > intmax())
18        if n == nP
19            T = v/u;
20        else
21            error('findLinearTransform:u','matrix too big to resolve');
22        end
23    else
24        A = zeros(nP*n,n*n);
25        for k = 1:nP
26            for i = ((k-1)*n+1):(k*n)
27                A(i,mod((((i-1)*n+1):(i*n))-1,n*n) + 1) = u(:,k)';
28            end
29        end
30        v = v(:);
31        T = reshape(A\v, n, n).';
32    end
33end
